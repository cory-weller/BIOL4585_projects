wget -L https://datadryad.org/bitstream/handle/10255/dryad.117072/data.black.widow.spiders.csv

#Change name of data
mv data.black.widow.spiders.csv data.csv

#In R
library(ggplot2)
library(data.table)

#Load data into R
data <- fread("data.csv")

#Plot 1
plot1 <- ggplot(data,aes(x=weight.at.boxing, y=web.weight.web, color=as.factor(attacked))) + geom_point() + labs(x="Weight of Black Widow", y="Weight of Web", title="Weight of Black Widow and Web")

#Plot 2
plot2 <- ggplot(data, aes(x=maturity.date, y=web.weight.web, color=as.factor(retreated))) + geom_point() + labs(x="Date of Maturity", y="Weight of Web", title="Date of Maturity and Weight of Web")

#Plot 3
plot3 <- ggplot(data, aes(x=own.web, y=struc.id, color=as.factor(attacked))) + geom_point() + labs(x="Spider's Web", y="Number of Lines on Web", title="Number of lines on Spider's Web") + geom_boxplot()

#Save plots
ggsave(plot1, file="SpiderWeight_vs_WebWeight.png")
ggsave(plot2, file="MaturityDate_vs_WebWeight.png")
ggsave(plot3, file="NumberWebLines_vs_OwnWeb.png")
